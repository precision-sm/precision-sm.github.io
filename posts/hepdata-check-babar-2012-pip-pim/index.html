  y_val.append(delem["value"])
  y_unc.append(delem["errors"][0]["symerror"])

##--- build data frame with table content
hd_115140_table1_df = pd.DataFrame(
  {"E lo": x_lo_val, "E hi": x_hi_val, "sigma val": y_val, "sigma unc": y_unc}
